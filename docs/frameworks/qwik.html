import { component$, useVisibleTask$, useSignal } from '@builder.io/qwik'; import { createCookieBanner } from 'smallest-cookie-banner'; export default component$(() => { const status = useSignal('Pending'); useVisibleTask$(() => { const banner = createCookieBanner({ msg: 'This site uses cookies.', onAccept: () => status.value = 'Accepted', onReject: () => status.value = 'Rejected', }); banner.show(); if (banner.status !== null) { status.value = banner.status ? 'Accepted' : 'Rejected'; } return () => banner.destroy(); }); return <p>Status: {status.value}</p>; });
